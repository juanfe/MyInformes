\begin{lstlisting}[caption=Ejemplo en Matlab]
function list = find_coordinates( matrix, value )
l = size(matrix);
coordenates_list = cell(l(1)*l(2),1);
total = 1;

% Evaluación de la matriz
for i=1:l(1)
    for j=1:l(2)
        if matrix(i,j)==value
            coordenates_list{total} = [i j];
            total = total + 1;
        end
    end
end

end
